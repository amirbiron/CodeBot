       sessionStorage.setItem('theme-mask-active', '1');
   }

   // ××—×¨×™ ×˜×¢×™× ×ª ×”×“×£
   if (sessionStorage.getItem('theme-mask-active') === '1') {
       sessionStorage.removeItem('theme-mask-active');
       mask.classList.add('active');
       setTimeout(() => mask.classList.remove('active'), 100);
   }
